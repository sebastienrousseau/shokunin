4//! Client-side search index generator.
5//!
6//! Generates a JSON search index and injects a search UI into HTML pages,
7//! providing instant full-text search without any server or external service.
8//!
9//! # How it works
10//!
11//! 1. At build time, `SearchIndex` scans all HTML files in the site directory.
12//! 2. It extracts the page title, URL, headings, and body text.
13//! 3. It writes a `search-index.json` file to the site root.
14//! 4. The `SearchPlugin` injects a `<script>` tag and search UI into every
15//!    HTML page that loads the index and performs client-side fuzzy matching.
16//!
17//! The search UI is a modal overlay activated by `Ctrl+K` / `Cmd+K`.
18

25/// A single entry in the search index.
26#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
27pub struct SearchEntry {
28    /// Page title extracted from `<title>` or first `<h1>`.
29    pub title: String,
30    /// Relative URL path (e.g., `/about/index.html`).
31    pub url: String,
32    /// Plain-text body content, truncated to `MAX_CONTENT_LENGTH`.
33    pub content: String,
34    /// Section headings found on the page.
35    pub headings: Vec<String>,
36}
37
38/// Maximum content length per page in the search index (characters).
39/// Keeps the index compact for fast client-side loading.
40pub const MAX_CONTENT_LENGTH: usize = 5_000;
41
42/// Maximum number of pages to index.
43pub const MAX_INDEX_ENTRIES: usize = 50_000;
44
45/// The complete search index written to `search-index.json`.
46#[derive(Debug, Clone, Serialize, Deserialize, Default)]
47pub struct SearchIndex {
48    /// All indexed pages.
49    pub entries: Vec<SearchEntry>,
50}
51
52impl SearchIndex {
53    /// Build a search index from all HTML files in `site_dir`.
54    ///
55    /// Walks the directory recursively, extracts content from each
56    /// `.html` file, and returns the populated index.
57    pub fn build(site_dir: &Path) -> Result<Self> {
58        let html_files = collect_html_files(site_dir)?;
59        let mut entries = Vec::with_capacity(html_files.len().min(MAX_INDEX_ENTRIES));
60
61        for path in html_files.iter().take(MAX_INDEX_ENTRIES) {
62            let html = fs::read_to_string(path)
63                .with_context(|| format!("cannot read {}", path.display()))?;
64
65            let rel_url = path
66                .strip_prefix(site_dir)
67                .unwrap_or(path)
68                .to_string_lossy()
69                .replace('\\', "/");
70
71            let title = extract_title(&html);
72            let headings = extract_headings(&html);
73            let content = extract_text(&html);
74
75            entries.push(SearchEntry {
76                title,
77                url: format!("/{rel_url}"),
78                content: truncate(&content, MAX_CONTENT_LENGTH),
79                headings,
80            });
81        }
82
83        Ok(Self { entries })
84    }
85
86    /// Write the index to `search-index.json` in the given directory.
87    pub fn write(&self, site_dir: &Path) -> Result<()> {
88        let json = serde_json::to_string(self)
89            .context("failed to serialize search index")?;
90        let path = site_dir.join("search-index.json");
91        fs::write(&path, json)
92            .with_context(|| format!("cannot write {}", path.display()))?;
93        Ok(())
94    }
95
96    /// Number of indexed pages.
97    pub fn len(&self) -> usize {
98        self.entries.len()
99    }
100
101    /// Returns true if the index has no entries.
102    pub fn is_empty(&self) -> bool {
103        self.entries.is_empty()
104    }
105}

107/// Plugin that generates a search index and injects client-side search UI.
108///
109/// # Example
110///
111/// ```rust
112/// use ssg::plugin::PluginManager;
113/// use ssg::search::SearchPlugin;
114///
115/// let mut pm = PluginManager::new();
116/// pm.register(SearchPlugin);
117/// ```
118#[derive(Debug, Copy, Clone)]
119pub struct SearchPlugin;
120
121impl Plugin for SearchPlugin {
122    fn name(&self) -> &str {
123        "search"
124    }
125
126    fn after_compile(&self, ctx: &PluginContext) -> Result<()> {
127        if !ctx.site_dir.exists() {
128            return Ok(());
129        }
130
131        let index = SearchIndex::build(&ctx.site_dir)?;
132        if index.is_empty() {
133            return Ok(());
134        }
135
136        index.write(&ctx.site_dir)?;
137
138        // Inject search UI into all HTML files
139        let html_files = collect_html_files(&ctx.site_dir)?;
140        for path in &html_files {
141            inject_search_ui(path)?;
142        }
143
144        println!(
145            "[search] Indexed {} pages, search-index.json written",
146            index.len()
147        );
148        Ok(())
149    }
150}

156/// Extract the page title from `<title>` tag or first `<h1>`.
157fn extract_title(html: &str) -> String {
158    // Try <title>
159    if let Some(start) = html.find("<title>") {
160        let after = &html[start + 7..];
161        if let Some(end) = after.find("</title>") {
162            let title = &after[..end];
163            if !title.trim().is_empty() {
164                return strip_tags(title).trim().to_string();
165            }
166        }
167    }
168    // Fallback to first <h1>
169    if let Some(start) = html.find("<h1") {
170        let after = &html[start..];
171        if let Some(gt) = after.find('>') {
172            let content = &after[gt + 1..];
173            if let Some(end) = content.find("</h1>") {
174                return strip_tags(&content[..end]).trim().to_string();
175            }
176        }
177    }
178    String::new()
179}
180
181/// Extract all heading text (`<h1>` through `<h6>`).
182fn extract_headings(html: &str) -> Vec<String> {
183    let mut headings = Vec::new();
184    for tag in &["h1", "h2", "h3", "h4", "h5", "h6"] {
185        let open = format!("<{tag}");
186        let close = format!("</{tag}>");
187        let mut search_from = 0;
188
189        while let Some(start) = html[search_from..].find(&open) {
190            let abs_start = search_from + start;
191            let after = &html[abs_start..];
192            if let Some(gt) = after.find('>') {
193                let content = &after[gt + 1..];
194                if let Some(end) = content.find(&close) {
195                    let text = strip_tags(&content[..end]).trim().to_string();
196                    if !text.is_empty() {
197                        headings.push(text);
198                    }
199                    search_from = abs_start + gt + 1 + end + close.len();
200                } else {
201                    break;
202                }
203            } else {
204                break;
205            }
206        }
207    }
208    headings
209}

211/// Extract visible text from HTML, stripping all tags.
212fn extract_text(html: &str) -> String {
213    // Remove <script>, <style>, <nav>, <header>, <footer> blocks
214    let mut clean = html.to_string();
215    for tag in &["script", "style", "nav", "header", "footer", "head"] {
216        let open = format!("<{tag}");
217        let close = format!("</{tag}>");
218        while let Some(start) = clean.find(&open) {
219            if let Some(end) = clean[start..].find(&close) {
220                clean.replace_range(start..start + end + close.len(), " ");
221            } else {
222                break;
223            }
224        }
225    }
226    strip_tags(&clean)
227}
228
229/// Remove all HTML tags, collapse whitespace.
230fn strip_tags(html: &str) -> String {
231    let mut result = String::with_capacity(html.len());
232    let mut in_tag = false;
233    for ch in html.chars() {
234        match ch {
235            '<' => in_tag = true,
236            '>' => {
237                in_tag = false;
238                result.push(' ');
239            }
240            _ if !in_tag => result.push(ch),
241            _ => {}
242        }
243    }
244    // Collapse whitespace
245    let mut collapsed = String::with_capacity(result.len());
246    let mut prev_space = false;
247    for ch in result.chars() {
248        if ch.is_whitespace() {
249            if !prev_space {
250                collapsed.push(' ');
251                prev_space = true;
252            }
253        } else {
254            collapsed.push(ch);
255            prev_space = false;
256        }
257    }
258    collapsed.trim().to_string()
259}

261/// Truncate a string to approximately `max` characters at a word boundary.
262fn truncate(s: &str, max: usize) -> String {
263    if s.chars().count() <= max {
264        return s.to_string();
265    }
266    let byte_pos: usize = s.char_indices()
267        .take(max)
268        .last()
269        .map(|(i, c)| i + c.len_utf8())
270        .unwrap_or(0);
271    let truncated = &s[..byte_pos];
272    if let Some(last_space) = truncated.rfind(' ') {
273        truncated[..last_space].to_string()
274    } else {
275        truncated.to_string()
276    }
277}

279/// Collect all `.html` files under `dir` (iterative, bounded).
280fn collect_html_files(dir: &Path) -> Result<Vec<PathBuf>> {
281    let mut files = Vec::new();
282    let mut stack = vec![dir.to_path_buf()];
283
284    while let Some(current) = stack.pop() {
285        if files.len() >= MAX_INDEX_ENTRIES {
286            break;
287        }
288        let entries = fs::read_dir(&current)
289            .with_context(|| format!("cannot read {}", current.display()))?;
290        for entry in entries {
291            let path = entry?.path();
292            if path.is_dir() {
293                stack.push(path);
294            } else if path.extension().map_or(false, |e| e == "html") {
295                files.push(path);
296            }
297        }
298    }
299
300    Ok(files)
301}

303/// Inject the search UI script into an HTML file.
304///
305/// Inserts a `<script>` block before `</body>` that:
306/// 1. Loads `search-index.json`
307/// 2. Creates a modal overlay with an input field
308/// 3. Performs case-insensitive substring matching on title + content
309/// 4. Displays results with highlighted snippets
310/// 5. Activates on `Ctrl+K` / `Cmd+K`
311fn inject_search_ui(path: &Path) -> Result<()> {
312    let html = fs::read_to_string(path)
313        .with_context(|| format!("cannot read {}", path.display()))?;
314
315    if html.contains("ssg-search-widget") {
316        return Ok(()); // Already injected
317    }
318
319    let script = SEARCH_WIDGET_SCRIPT;
320
321    let injected = if let Some(pos) = html.rfind("</body>") {
322        format!("{}{}{}", &html[..pos], script, &html[pos..])
323    } else {
324        format!("{html}{script}")
325    };
326
327    fs::write(path, injected)
328        .with_context(|| format!("cannot write {}", path.display()))?;
329    Ok(())
330}
